export type Topic = 'addition' | 'subtraction' | 'word-problems'

export interface LessonStep {
  text: string
  narrationAudio: string   // '/audio/lessons/{lessonId}/step-{N}.mp3'
  equation?: string        // optional Unicode e.g. '3 + 4 = ?'
}

export interface WorkedExample {
  steps: LessonStep[]
}

export interface Lesson {
  id: string               // slug: 'addition-grade1-01'
  title: string
  grade: 1 | 2 | 3
  topic: Topic
  order: number
  ccStandards: string[]    // e.g. ['1.OA.A.1', '1.OA.C.6']
  workedExample: WorkedExample
}

// Discriminated union
// MultipleChoiceProblem: type 'multiple-choice', choices: number[] (always 4, unordered)
// DigitGridProblem: type 'digit-grid', no choices field
export type Problem = MultipleChoiceProblem | DigitGridProblem

export interface CurriculumData {
  lessons: Lesson[]
  problems: Problem[]
}
